FROM 0.* to 0.5

 - The method addcolumn is now called updatecolumn           
 - You can only set data trough Zend_Db_Select object
 - You must first set the query and then use the updateColumn method          
 - You MUST set fields alias like this:
 [CODE]
 $db->select()->from('table',array('alias'=>'field'));
 [/CODE]
 AND NOT
 [CODE]
 $db->select()->from('table',array('field as alias'));
 [/CODE]
